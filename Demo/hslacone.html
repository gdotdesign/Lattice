HSLA Color picker

Javascript 

        
Pickers.Color.attach($("cpick"));
$("cpick").addEvent('change', function(e){
  this.setStyle('background-color',e.color)
  if(e.color.hsb[2] < 50)
    this.setStyle('color','#fff');
  else
    this.setStyle('color','#333');
});
